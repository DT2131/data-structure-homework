#include <bits/stdc++.h>
using namespace std;
const int MAXN=1e5+10;
template <class Type>
class Queue{
    Type elem[MAXN];
    int head=-1,tail=-1;
public:
    void clear(){
        head=tail=-1;
    }
    void push(Type str){
        elem[++tail]=str;
    }
    Type front(){
        return elem[++head];
    }
    bool empty(){
        return head==tail?1:0;
    }
    void pop(){
        head++;
    }
    void traval(){
        for(int i=head+1;i<=tail;i++)
            cout<<elem[i]<<endl;
    }
};
template <class Type>
class Stack{
    Type elem[MAXN];
    int head=-1;
public:
    void push(Type str){
        elem[++head]=str;
    }
    bool empty(){
        return head==-1?:0;
    }
    Type top(){
        return elem[head];
    }
    void pop(){
        head--;
    }
    void traval(){
        for(int i=head;i>=0;i--) cout<<elem[i]<<endl;
    }
    void clear(){
        head=-1;
    }
};

Queue <string> sque;
Queue <double> dque;
Stack <string> ssta;
Stack <double> dsta;
int pri[128]={0};
void divide(string str){
    str+="@";
    int len=str.size();
    bool innum=0;bool cut=0;
    double num=0;int cnt=1;
    for(int i=0;i<len;i++){
        if(str[i]<='9'&&str[i]>='0'){
            innum=1;
            num*=10;
            num+=str[i]-'0';
            if(cut) cnt*=10;
        }else if(str[i]=='.'){
            cut=1;
        }else{
            if(innum){dque.push(num/cnt);innum=0;cnt=1;cut=0;num=0;}
            string temp;
            if(str[i]=='+'||str[i]=='-'||str[i]=='/'||str[i]=='*'||str[i]=='('||str[i]==')'||str[i]=='^'||str[i]=='@'){
                temp=str[i];
            }else if(str[i]=='l'){
                temp="ln";
                i++;
            }else if(str[i]=='s'){
                temp="sqrt";
                i+=3;
            }
            sque.push(temp);
        }
    }
}
